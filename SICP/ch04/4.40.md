		(define (multiple-dwelling-1)
		 (let ((baker (amb 1 2 3 4))
			   (cooper (amb 2 3 4 5))
			   (miler (amb 1 2 3 4 5))
			   (fletcher (amb 2 3 4))
			   (smith (amb 1 2 3 4 5)))
		  (require (> miler cooper))
		  (require (not (= (abs (- fletcher cooper)) 1)))
		  (require (not (= (abs (- smith fletcher)) 1)))
		  (require (distinct? (list baker cooper fletcher miler smith)))))
